# Condicionais

## if

if(resultado booleano){
    se for verdadeiro
}
se for falso


## else

if(resultado booleano){
    se for verdadeiro
}else{
    se for falso
}

## else if

if(resultado booleano){
    se for verdadeiro
}else if(resultado booleano){
    se for verdadeiro
}else{
    resultado falso
}
